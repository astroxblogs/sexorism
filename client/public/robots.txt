// app/robots.ts
import type { MetadataRoute } from 'next'

export default function robots(): MetadataRoute.Robots {
  return {
    rules: [
      {
        userAgent: '*',
        allow: '/',
        disallow: ['/cms/', '/admin/', '/api/', '/_next/'],
      },
      // If you still want per-bot groups, REPEAT the disallows for each:
      // { userAgent: 'Googlebot', allow: '/', disallow: ['/cms/', '/admin/', '/api/', '/_next/'] },
      // { userAgent: 'Bingbot',   allow: '/', disallow: ['/cms/', '/admin/', '/api/', '/_next/'] },
    ],
    sitemap: 'https://innvibs.com/sitemap.xml',
    // (Optionally serve a different sitemap on innvibs.in if needed)
  }
}
